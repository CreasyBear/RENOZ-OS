# Wireframe: DOM-SET-006c - Custom Fields Form Component

## Story Reference

- **Story ID**: DOM-SET-006c
- **Name**: Custom Fields Form Component
- **PRD**: memory-bank/prd/domains/settings.prd.json
- **Type**: Reusable Form Section Component
- **Component Type**: Dynamic form renderer for custom fields

## Overview

Reusable component that dynamically renders custom fields based on entity type. Renders appropriate input types (text, number, date, select, checkbox), validates required fields, and integrates with parent forms for both create and edit modes.

## UI Patterns (Reference Implementation)

### Form (Dynamic Field Renderer)
- **Pattern**: RE-UI Form
- **Reference**: `_reference/.reui-reference/registry/default/ui/form.tsx`
- **Features**:
  - Dynamic form that renders based on custom field configuration
  - Field-level validation with error messages
  - Integration with React Hook Form for parent form state

### Input (Text and Number Fields)
- **Pattern**: RE-UI Input
- **Reference**: `_reference/.reui-reference/registry/default/ui/input.tsx`
- **Features**:
  - Text input for string custom fields
  - Number input with min/max validation for numeric fields
  - Help text below showing field description

### Select (Dropdown Custom Fields)
- **Pattern**: RE-UI Select
- **Reference**: `_reference/.reui-reference/registry/default/ui/select.tsx`
- **Features**:
  - Dropdown for select-type custom fields
  - Options defined in custom field configuration
  - Searchable when many options exist

### Checkbox (Boolean Custom Fields)
- **Pattern**: RE-UI Checkbox
- **Reference**: `_reference/.reui-reference/registry/default/ui/checkbox.tsx`
- **Features**:
  - Checkbox for boolean custom fields
  - Label and description text from field configuration
  - Controlled state from parent form

### Card (Field Group Container)
- **Pattern**: RE-UI Card
- **Reference**: `_reference/.reui-reference/registry/default/ui/card.tsx`
- **Features**:
  - Card container grouping custom fields section
  - Header showing "Custom Fields" title
  - Collapsible on mobile to save space

## Accessibility Notes

### Focus Order

1. **Section Focus**
   - Tab into custom fields section
   - Fields receive focus in sortOrder sequence

2. **Field Navigation**
   - Tab through each field in order
   - Required fields announced
   - Error fields announced

3. **Select Dropdown**
   - Space/Enter opens
   - Arrow keys navigate options
   - Enter/Space selects
   - Escape closes

4. **Date Picker**
   - Focus calendar button
   - Arrow keys navigate dates
   - Enter selects date

5. **Checkbox**
   - Space toggles

## Animation Choreography

### Section Load

```
FIELDS ENTRY:
- Duration: 400ms total
- Stagger: 50ms per field
- Each field: translateY(8px) -> translateY(0)
- Opacity: 0 -> 1
```

### Field Interactions

```
FOCUS:
- Duration: 150ms
- Border: transition to primary color
- Background: subtle highlight

DROPDOWN OPEN:
- Duration: 200ms
- Easing: ease-out
- Transform: translateY(-4px) -> translateY(0)
- Opacity: 0 -> 1

DATE PICKER OPEN:
- Duration: 200ms
- Scale: 0.95 -> 1
- Opacity: 0 -> 1
```

### Validation

```
ERROR SHAKE:
- Duration: 400ms
- Transform: translateX sequence (-4, 4, -4, 4, 0)

ERROR MESSAGE ENTRY:
- Duration: 200ms
- Transform: translateY(-4px) -> translateY(0)
- Opacity: 0 -> 1
- Color: fade to error color

ERROR CLEAR:
- Duration: 150ms
- Opacity: 1 -> 0
```

### Checkbox

```
CHECK TOGGLE:
- Duration: 150ms
- Checkmark scale: 0 -> 1.1 -> 1
- Background: transition color
```

  fieldType: 'text' | 'number' | 'date' | 'select' | 'checkbox';
  options: string[] | null;
  isRequired: boolean;
  sortOrder: number;
}

// Hook for custom fields
interface UseCustomFieldsOptions {
  entityType: EntityType;
  entityId?: string;
}

interface UseCustomFieldsReturn {
  // Field definitions
  fields: CustomFieldDefinition[];
  isLoadingFields: boolean;
  fieldsError: Error | null;

  // Field values (for edit mode)
  values: CustomFieldValues;
  isLoadingValues: boolean;
  valuesError: Error | null;

  // Validation
  validate: () => boolean;
  errors: Record<string, string>;

  // Update values
  setValue: (fieldId: string, value: CustomFieldValue) => void;
  setValues: (values: CustomFieldValues) => void;

  // Reset
  reset: () => void;
}

// Field renderer props
interface CustomFieldInputProps {
  field: CustomFieldDefinition;
  value: CustomFieldValue | undefined;
  onChange: (value: CustomFieldValue) => void;
  error?: string;
  disabled?: boolean;
}

// Validation schema builder
function buildCustomFieldSchema(
  fields: CustomFieldDefinition[]
): z.ZodObject<Record<string, z.ZodTypeAny>> {
  const shape: Record<string, z.ZodTypeAny> = {};

  for (const field of fields) {
    let schema: z.ZodTypeAny;

    switch (field.fieldType) {
      case 'text':
        schema = z.string();
        break;
      case 'number':
        schema = z.number();
        break;
      case 'date':
        schema = z.string().refine(isValidDate);
        break;
      case 'select':
        schema = z.enum(field.options as [string, ...string[]]);
        break;
      case 'checkbox':
        schema = z.boolean();
        break;
    }

    if (!field.isRequired) {
      schema = schema.optional();
    }

    shape[field.id] = schema;
  }

  return z.object(shape);
}

## Integration Example

```typescript
// In customer form
import { CustomFieldsFormSection } from '@/components/shared/custom-fields-form-section';
import { useCustomFields } from '@/hooks/use-custom-fields';

function CustomerCreateForm() {
  const form = useForm({
    defaultValues: {
      name: '',
      email: '',
      // ... standard fields
    },
  });

  const customFields = useCustomFields({
    entityType: 'customers',
  });

  const handleSubmit = async (data: FormData) => {
    // Validate custom fields
    if (!customFields.validate()) {
      return;
    }

    // Save customer with custom field values
    await createCustomer({
      ...data,
      customFieldValues: customFields.values,
    });
  };

  return (
    <Form {...form} onSubmit={handleSubmit}>
      {/* Standard fields */}
      <FormSection title="Basic Information">
        <FormField name="name" label="Business Name" required />
        <FormField name="email" label="Email" required />
      </FormSection>

      {/* Custom fields section */}
      {customFields.fields.length > 0 && (
        <CustomFieldsFormSection
          entityType="customers"
          values={customFields.values}
          onChange={customFields.setValues}
          errors={customFields.errors}
          isLoading={customFields.isLoadingFields}
          columns={2}
        />
      )}

      <FormFooter>
        <Button type="submit">Create Customer</Button>
      </FormFooter>
    </Form>
  );
}
```
